SELECT ao.id,ao.created,INS.installer_name,IF ((CASE WHEN 1=1 THEN (
                SELECT COUNT(0) AS CNT
                FROM applyonlin_docs
                WHERE applyonlin_docs.application_id = ao.id
                AND applyonlin_docs.doc_type = 'profile' 
) END) = 1, 'Yes', 'No') AS Profile_Photo,(CASE WHEN 1=1 THEN (
                SELECT COUNT(0) AS CNT
                FROM apply_onlines_errors_log
                WHERE apply_onlines_errors_log.application_id = ao.id
) END) AS count_error FROM `apply_onlines` as ao INNER JOIN installers as INS ON INS.id = ao.installer_id  WHERE `application_status` not IN ('99','29','30','22')  order by Profile_Photo asc
